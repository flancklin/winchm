无限参数函数1

function sum(...$numbers) {
    $acc = 0;
    foreach ($numbers as $n) {
        $acc += $n;
    }
    return $acc;
}

echo sum(1, 2, 3, 4);

{
    echo func_num_args();      //参数个数【3】

函数引用 从函数返回一个引用

function &func()
{
    static  $p = 99;
    return $p;//['p' => $p];
}
$res = &func();     //等同于  $res = &$p;切记申明的时候有&，调用的时候也有&
echo $res;            //99
$res = 88;
echo func();         //88
