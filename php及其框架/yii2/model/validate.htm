//方式一 对象属性逐一赋值  先赋值再验证
    $obj = new Model();
    $obj->name = "张三";
    $obj->age = 12;
    if(!$obj->validate()){
        var_dump($obj->errors);//errors是二维数组
    }
//方式二  块赋值       先赋值再验证
    $obj = new Model();
    $obj->attributes = ['name' => '张三', 'age' => 12];
    if(!$obj->validate()){
        var_dump($obj->errors);//errors是二维数组
    }
//方式三  错误的写法
    $data = ['name' => '张三', 'age' => 12];
    $obj = new Model();
    if(!$obj->validate($data)){
        var_dump($obj->errors);//errors是二维数组
    }
    $obj->attributes = $data;
