            function sortByField(data, field,order) {
                return  data.sort(function (a, b) {
                    if(typeof(a[field]) == 'undefined' || b[field] == 'undefined' ) return 0;
                    var value1 = a[field],
                        value2 = b[field],
                        balance = value1 - value2;
                    if(order == 'desc') return  -balance;
                    if(order == 'asc') return  balance;
                    return  0;
                });
            }
            var aa = sortByField(newArray, 'value', 'desc');
